// 注册用户
POST /api/register form-data
    参数 : 
        name  3-18位 字母数字 '-' '_'
        password    3-18位字母数字 '-' '_'
        email   必须包含@qq.com
    返回值 : json {
        code:   0 02 03 04 05
        message:
    }

// 登录用户
GET /api/login  uri
    参数 : 
        account  用户名或者邮箱
        password    密码 (需要经过md5加密)
    返回值 : json {
        code:   0 02 03 06 07 08
        message:
    }
    设置 Cookie "frostAuth"
